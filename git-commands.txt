# Команды для загрузки на GitHub
# Выполняй по порядку в терминале (в папке с ботом)

# 1. Инициализация Git
git init

# 2. Добавить все файлы
git add .

# 3. Создать коммит
git commit -m "Initial commit: Telegram Anti-Bot Protection"

# 4. Переименовать ветку в main
git branch -M main

# 5. Добавить удалённый репозиторий
git remote add origin https://github.com/dimasik47k-ship-it/telegram-antibot.git

# 6. Загрузить на GitHub
git push -u origin main

# Готово! Файлы загружены на GitHub
